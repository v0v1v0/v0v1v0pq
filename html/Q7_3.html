## S3 method for class 'Q7instance'
clone(inst, deep = TRUE, ...)

Arguments

to copy nested object instances recursively; Boolean
...

dot-dot-dot

Value

Q7 object instance

Examples


Type1 <- type(function(num){
  print_num <- function(){
    base::print(num)
  }
})
myType1 <- Type1(1)
myType1$print_num()
myType1_clone <- clone(myType1)
myType1_clone$print_num()
